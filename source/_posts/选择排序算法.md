### 选择排序算法C语言实现
```cpp
//选择排序算法
void sort(int list[], int n)
{
	for (int i = 0; i < n - 1; i++)
	{
	//	int min = i;
		for (int j = i + 1; j < n;j++)
		if (list[j] < list[i])
	//		min = j;
		select_swap(list+j, list+i);
	}
}

void select_swap(int *a, int *b)
{
	int temp;
	temp = *a;
	*a = *b;
	*b = temp;
}

```
